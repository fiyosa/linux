docker network create proxy

# run 
docker run -d \
  --name nginx-proxy \
  --restart always \
  -p 80:80 \
  -p 443:443 \
  -v ~/docker/nginx/conf.d:/etc/nginx/conf.d \
  -v ~/docker/certbot/www:/var/www/certbot \
  -v ~/docker/certbot/conf:/etc/letsencrypt \
  --network proxy \
  nginx:alpine
  
# set network all containers to proxy

# reload nginx-proxy after edit folder ~/docker/nginx/conf.d
docker exec nginx-proxy nginx -s reload

# portfolio.conf
server {
    listen 80;
    server_name fiyosa.me www.fiyosa.me;

    location /.well-known/acme-challenge/ {
        root /var/www/certbot;
    }

    location / {
        proxy_pass http://portfolio:8000;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
    }
}

# portfolio_ssl.conf
server {
    listen 80;
    server_name fiyosa.me www.fiyosa.me;
    location /.well-known/acme-challenge/ {
        root /var/www/certbot;
    }
    return 301 https://$host$request_uri;
}
server {
    listen 443 ssl;
    server_name fiyosa.me www.fiyosa.me;
    ssl_certificate /etc/letsencrypt/live/fiyosa.me/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/fiyosa.me/privkey.pem;
    location / {
        proxy_pass http://portfolio:8000;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
    }
}

# conf.d/basic-auth.conf

server {
    listen 80;
    server_name anjay.fiyosa.me;

    auth_basic "Restricted";
    auth_basic_user_file /etc/nginx/.htpasswd;

    location / {
        proxy_pass http://anjay:3000;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection "upgrade";
    }
}

# conf.d/dbgate.conf

server {
    listen 80;
    server_name dbgate.fiyosa.me;

    auth_basic "Restricted";
    auth_basic_user_file /etc/nginx/.htpasswd;

    location / {
        auth_basic "Restricted";
        auth_basic_user_file /etc/nginx/.htpasswd;
        proxy_pass http://dbgate:3000;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection "upgrade";
    }

    location /config {
        auth_basic off;
        proxy_pass http://dbgate:3000;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
    }

    location /stream {
        auth_basic off;
        proxy_pass http://dbgate:3000;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection "upgrade";
    }
}
